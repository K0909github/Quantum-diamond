#92atoms_3eV_Ar_implantation_to_Si_SW_potential_filedata.txt
#Newton's third law
newton on

units metal
atom_style atomic
dimension 3
boundary p p p

#setting_simulation_box
region          mybox0 block -500 500 -500 500 -500 500
create_box      3 mybox0 #2types atom

# Neighbor list settings 
neighbor        2.0 bin 
neigh_modify    every 1 delay 0 check yes

#set_up atom
read_data       1000K_annealed_silicon_320A.data add append
group           Si_sub type 1
group           Fixed_sub type 2

#Ar-cluster formation
region          mybox2 sphere 0 0 100 9.73 units box   #92atoms
lattice         fcc 5.4051
create_atoms    3 region mybox2
mass 3          39.95 #Ar
group           Ar_cluster type 3

#Initialization
velocity        Si_sub create 300 12345 mom yes dist gaussian #type 1
velocity        Fixed_sub set 0.0 0.0 0.0                     #type 2
velocity        Ar_cluster set 0 0 -37.97 units box              #type 3, 3eV

#Force field 
pair_style      hybrid sw/kk zbl/kk 4.5 4.5 lj/cut/kk 8.0
pair_coeff      * * sw/kk Si.sw Si Si NULL
pair_coeff      1 3 zbl/kk 14 18
pair_coeff      2 3 zbl/kk 14 18
pair_coeff      3 3 lj/cut/kk 0.01032 3.405

neigh_modify delay 0


#Operation
timestep 0.001

fix             hold Fixed_sub setforce 0.0 0.0 0.0
fix             f1 all nve

#Output
thermo          1000 #Display by 100 timestep
thermo_style    custom time ke pe etotal temp

dump            Ar_cluster_to_Sisub all xyz 50 92atoms_3eV_Ar_cluster_to_Sisub.xyz
dump_modify     Ar_cluster_to_Sisub element Si_Sub Fixed_Si Ar_cluster

#Actions
run             12000

undump          Ar_cluster_to_Sisub
write_data      92atoms_3eV_Ar-cluster_to_Si.data
